    usage = game

    #KOR Pickelhaube

    COM_KOR_war_headgear = {
        dna_modifiers = {
            accessory = {
                mode = add
                gene = com_headgear
                template = com_KOR_headgear
                range = { 0 1 } # For the randomness to work correctly
            }

            accessory = {
                mode = add
                gene = headgear
                template = no_headgear
                range = { 0 1 } # For the randomness to work correctly
            }

        weight = {
            base = 0
            modifier = {
                add = 1000
                exists = scope:character
                scope:character ?= {
                    monarchy_clothes_trigger = yes
                    is_female = no
                    age >= 10
                    has_role = general
                    owner ?= {
                        has_variable = KOR_korean_empire_declared
                        OR = {
                            c:KOR ?= this
                            was_formed_from = KOR
                        }
                    }
                }
            }
            # WCR compatibility
            modifier = {
                add = 1000
                COM_WCR_enable_outfit = yes
                scope:character ?= {
                    has_role = general
                    owner ?= {
                        OR = {
                            c:KOR ?= this
                            was_formed_from = KOR
                        }
                    }
                }
            }
            modifier = {
                add = -1000
                COM_WCR_disable_outfit = yes
                scope:character ?= {
                    has_role = general
                    owner ?= {
                        OR = {
                            c:KOR ?= this
                            was_formed_from = KOR
                        }
                    }
                }
            }
        }
    }

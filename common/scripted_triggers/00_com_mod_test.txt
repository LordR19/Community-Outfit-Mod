COM_is_early_military = {
    OR = {
        AND = {
            exists = scope:character
            scope:character ?= {
                owner ?= {
                    NOT = {
                        has_technology_researched = bolt_action_rifles
                    }
                }
            }
        }
        AND = {
            exists = scope:pop
            scope:pop ?= {
                owner ?= {
                    NOT = {
                        has_technology_researched = bolt_action_rifles
                    }
                }
            }
        }
    }
}

COM_is_mid_military = {

COM_is_monarch_leader = {
    exists = scope:character
    scope:character ?= {
        monarchy_clothes_trigger = yes
        is_female = no
        age >= define:NCharacters|ADULT_AGE
    }
}

COM_is_republic_leader = {
    exists = scope:character
    scope:character ?= {
        owner ?= {
            coa_republic_trigger = yes
        }
        is_ruler = yes
        age >= define:NCharacters|ADULT_AGE
    }
}

COM_is_general_leader = {
    exists = scope:character
    scope:character ?= {
        OR = {
            AND = {
                has_role = politician
                is_ruler = no
                is_heir = no
                is_interest_group_type = ig_armed_forces
                mr_com_special_character_blocker = yes
            }
            has_role = general
        }
    }
}

### TAG triggers

    # checks whether your tag is, was formed from or is a vassal of the tag in curly brackets like "COM_is_or_was_tag = { TAG = AUS}" for Austria
COM_is_or_was_tag = {
    OR = {
        scope:character ?= {
            home_country ?= {
                OR = {
                    AND = {
                        c:$TAG$ ?= this
                        is_subject = no # preventing uniform conflicts if AUS becomes a subject of another country with puppet uniforms
                    }
                    top_overlord = {
                        was_formed_from = $TAG$
                    }
                    AND = {
                        exists = c:$TAG$
                        is_subject_of = c:$TAG$
                    }
                    was_formed_from = $TAG$
                }
            }
        }
        scope:pop ?= {
            home_country ?= {
                OR = {
                    AND = {
                        c:$TAG$ ?= this
                        is_subject = no # preventing uniform conflicts if AUS becomes a subject of another country with puppet uniforms
                    }
                    top_overlord = {
                        was_formed_from = $TAG$
                    }
                    AND = {
                        exists = c:$TAG$
                        is_subject_of = c:$TAG$
                    }
                    was_formed_from = $TAG$
                }
            }
        }
    }
}

COM_is_swedish_SCA = {
    exists = scope:character
    scope:character ?= {
        has_culture = cu:swedish
        OR = {
            AND = {
                has_role = politician
                is_ruler = no
                is_interest_group_type = ig_armed_forces
            }
            has_role = general
            has_role = admiral
        }
        owner ?= {
            NOT = {
                has_technology_researched = bolt_action_rifles
            }
            OR = {
                c:SCA ?= this
                was_formed_from = SCA
            }
        }
    }
}
